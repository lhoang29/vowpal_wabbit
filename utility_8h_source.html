100  static inline uint32_t fmix(uint32_t h)
101  {
102  h ^= h >> 16;
103  h *= 0x85ebca6b;
104  h ^= h >> 13;
105  h *= 0xc2b2ae35;
106  h ^= h >> 16;
107 
108  return h;
109  }

113  uint32_t uniform_hash(const void * key, size_t len, uint32_t seed)
114  {
115  const uint8_t * data = (const uint8_t*)key;
116  const int nblocks = (int)len / 4;
117 
118  uint32_t h1 = seed;
119 
120  const uint32_t c1 = 0xcc9e2d51;
121  const uint32_t c2 = 0x1b873593;
122 
123  // --- body
124  const uint32_t * blocks = (const uint32_t *)(data + nblocks * 4);
125 
126  for (int i = -nblocks; i; i++) {
127  uint32_t k1 = getblock(blocks, i);
128 
129  k1 *= c1;
130  k1 = ROTL32(k1, 15);
131  k1 *= c2;
132 
133  h1 ^= k1;
134  h1 = ROTL32(h1, 13);
135  h1 = h1 * 5 + 0xe6546b64;
136  }
137 
138  // --- tail
139  const uint8_t * tail = (const uint8_t*)(data + nblocks * 4);
140 
141  uint32_t k1 = 0;
142 
143  switch (len & 3) {
144  case 3: k1 ^= tail[2] << 16;
145  case 2: k1 ^= tail[1] << 8;
146  case 1: k1 ^= tail[0];
147  k1 *= c1; k1 = ROTL32(k1, 15); k1 *= c2; h1 ^= k1;
148  }
149 
150  // --- finalization
151  h1 ^= len;
152 
153  return fmix(h1);
154  }

191  template<bool trailing_zeros>
192  static void print_mantissa(char*& begin, float f)
193  { // helper for print_float
194  char values[10];
195  size_t v = (size_t)f;
196  size_t digit = 0;
197  size_t first_nonzero = 0;
198  for (size_t max = 1; max <= v; ++digit)
199  {
200  size_t max_next = max * 10;
201  char v_mod = (char) (v % max_next / max);
202  if (!trailing_zeros && v_mod != '\0' && first_nonzero == 0)
203  first_nonzero = digit;
204  values[digit] = '0' + v_mod;
205  max = max_next;
206  }
207  if (!trailing_zeros)
208  for (size_t i = max_digits; i > digit; i--)
209  *begin++ = '0';
210  while (digit > first_nonzero)
211  *begin++ = values[--digit];
212  }

214  static void print_float(char* begin, float f)
215  {
216  bool sign = false;
217  if (f < 0.f)
218  sign = true;
219  float unsigned_f = fabsf(f);
220  if (unsigned_f < max_float && unsigned_f > min_float)
221  {
222  if (sign)
223  *begin++ = '-';
224  print_mantissa<true>(begin, unsigned_f);
225  unsigned_f -= (size_t)unsigned_f;
226  unsigned_f *= max_int;
227  if (unsigned_f >= 1.f)
228  {
229  *begin++ = '.';
230  print_mantissa<false>(begin, unsigned_f);
231  }
232  }
233  else if (unsigned_f == 0.)
234  *begin++ = '0';
235  else
236  {
237  sprintf(begin, "%g", f);
238  return;
239  }
240  *begin = '\0';
241  return;
