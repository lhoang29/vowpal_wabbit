377  std::tuple<u32, float, bool> Choose_Action(u64 salted_seed, Ctx& context)
378  {
379  PRG::prg random_generator(salted_seed);
380 
381  // Invoke the default scorer function
382  vector<float> scores = m_default_scorer.Score_Actions(context);
383  u32 num_scores = (u32)scores.size();
384  if (num_scores != m_num_actions)
385  {
386  throw std::invalid_argument("The number of scores returned by the scorer must equal number of actions");
387  }
388 
389  u32 i = 0;
390 
391  float max_score = -FLT_MAX;
392  for (i = 0; i < num_scores; i++)
393  {
394  if (max_score < scores[i])
395  {
396  max_score = scores[i];
397  }
398  }
399 
400  // Create a normalized exponential distribution based on the returned scores
401  for (i = 0; i < num_scores; i++)
402  {
403  scores[i] = exp(m_lambda * (scores[i] - max_score));
404  }
405 
406  // Create a discrete_distribution based on the returned weights. This class handles the
407  // case where the sum of the weights is < or > 1, by normalizing agains the sum.
408  float total = 0.f;
409  for (size_t i = 0; i < num_scores; i++)
410  total += scores[i];
411 
412  float draw = random_generator.Uniform_Unit_Interval();
413 
414  float sum = 0.f;
415  float action_probability = 0.f;
416  u32 action_index = num_scores - 1;
417  for (u32 i = 0; i < num_scores; i++)
418  {
419  scores[i] = scores[i] / total;
420  sum += scores[i];
421  if (sum > draw)
422  {
423  action_index = i;
424  action_probability = scores[i];
425  break;
426  }
427  }
428 
429  // action id is one-based
430  return std::tuple<u32, float, bool>(action_index + 1, action_probability, true);
431  }
